## 【自定义异常以及全局异常处理类】
### ResultEnum
```java

import lombok.Getter;

/**
 * 异常编码定义规则：
 * 系统管理模块： 1000~9999
 * 其它业务模块： 10000~19999
 */
public enum ResultEnum {

    /*
     * 成功
     */
    SUCCESS(200, "操作成功"),
    /*
     * 失败
     */
    FAIL(-1, "fail"),
    FEIGN_CALL_FAIL(-2, "调用服务[%s]失败"),
    UPDATE_DATA_FAIL(1001, "更新数据库失败"),
    INSERT_DATA_FAIL(1002, "插入数据库失败"),
    SELECT_DATA_FAIL(1003, "没有查到数据"),
    FILE_QUERY_ERROR(1004,"文件查询异常"),
    FILE_UPLOAD_FAIL(1005, "上传文件失败"),
    FILE_DOWNLOAD_FAIL(1006, "下载文件失败"),
    PARAM_ERROR(1007, "参数错误：[%s]");

    @Getter
    private final Integer code;

    @Getter
    private final String message;

    ResultEnum(Integer code, String message) {
        this.code = code;
        this.message= message;
    }
}

```

### GlobalException
```java

/**
 * 全局异常
 */
public class GlobalException extends RuntimeException{

    protected Integer code;

    public GlobalException(String message, Integer code) {
        super(message);
        this.code = code;
    }

    public GlobalException(ResultEnum resultEnum) {
        super(resultEnum.getMessage());
        this.code = resultEnum.getCode();
    }

    public void setCode(Integer code) {
        this.code = code;
    }

    public Integer getCode() {
        return this.code;
    }
}

```
### BusinessException 
```java

/**
 * 统一业务异常定义
 */
public class BusinessException extends GlobalException {
    private BusinessException(String message, Integer code) {
        super(message, code);
    }

    private BusinessException(ResultEnum resultEnum) {
        super(resultEnum);
    }

    public static BusinessException bizException(String message, Integer code) {
        return new BusinessException(message, code);
    }

    public static BusinessException bizException(ResultEnum resultEnum) {
        return new BusinessException(resultEnum);
    }

    public static BusinessException bizException(ResultEnum resultEnum, String... param) {
        return new BusinessException(String.format(resultEnum.getMessage(), param), resultEnum.getCode());
    }

    public static BusinessException bizException(String errorMessage, ResultEnum resultEnum) {
        return new BusinessException(errorMessage, resultEnum.getCode());
    }

    public static BusinessException bizException(String errorMessage, ResultEnum resultEnum, String... param) {
        return new BusinessException(String.format(resultEnum.getMessage(), param) + "，" + errorMessage, resultEnum.getCode());
    }
}

```
